				<!--请保证iframe id 与地址中 ifname值相同-->
				<iframe src="{iframesrc}?type=iframe&fromaddr=snifeng&ifname={ifname}" 
					id="{ifname}" 
					name="{ifname}" 
					scrolling="no"
					frameborder="no" 
					style="border: 0;width:100%;height:500px;">
				</iframe>
				<!--
				设置iframe功能,可传递参数：
					height 		禁止自动高度
					fromaddr	禁止记录来源
				传递多个参数用&分隔，默认全开，传递后禁止此功能。如：{iframeheightsrc}/height&fromaddr  
				//-->
				<script src="{iframeheightsrc}"></script>
